
CREATE DATABASE IF NOT EXISTS alliance;

CREATE TABLE IF NOT EXISTS `alliance`.`category` (
  `id` INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  `name` VARCHAR(150) NOT NULL,
  PRIMARY KEY (`id`)
)
ENGINE = InnoDB;

CREATE TABLE `alliance`.`view` (
  `id` INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  `type` INTEGER UNSIGNED NOT NULL,
  `category` INTEGER UNSIGNED NOT NULL DEFAULT 0,
  `title` VARCHAR(250) NOT NULL,
  `content` TEXT NOT NULL,
  PRIMARY KEY (`id`)
)
ENGINE = InnoDB;

ALTER TABLE `alliance`.`view` DROP COLUMN `type`;

CREATE TABLE IF NOT EXISTS `alliance_space`.`page` (
  `id` INTEGER UNSIGNED NOT NULL AUTO_INCREMENT,
  `title` VARCHAR(250) NOT NULL,
  `content` TEXT NOT NULL,
  PRIMARY KEY (`id`)
)
ENGINE = InnoDB;
